**1. src/lib/auth/ad-inquiry-action.ts 생성**

보안 원칙:
- 인증 불필요 (공개 문의 폼)
- zod 검증 → 환경변수 확인 → Resend 발송 순서
- `RESEND_API_KEY`, `OPERATOR_EMAIL` 미설정 시 graceful error (앱 크래시 없음)

```typescript
'use server'

import { z } from 'zod'
import { Resend } from 'resend'

const inquirySchema = z.object({
  company: z.string().min(1, '업체명을 입력해 주세요').max(100, '업체명은 100자 이하로 입력해 주세요'),
  contact: z.string().min(1, '연락처를 입력해 주세요').max(200, '연락처는 200자 이하로 입력해 주세요'),
  period:  z.enum(['7', '30', '90'], { errorMap: () => ({ message: '광고 기간을 선택해 주세요' }) }),
  message: z.string().min(1, '문의 내용을 입력해 주세요').max(1000, '문의 내용은 1000자 이하로 입력해 주세요'),
})

const PERIOD_LABEL: Record<string, string> = {
  '7':  '1주(7일)',
  '30': '1달(30일)',
  '90': '3달(90일)',
}

export async function submitAdInquiry(
  formData: FormData,
): Promise<{ error: string | null }> {
  // 1. zod 검증
  const raw = {
    company: formData.get('company') ?? '',
    contact: formData.get('contact') ?? '',
    period:  formData.get('period') ?? '',
    message: formData.get('message') ?? '',
  }
  const parsed = inquirySchema.safeParse(raw)
  if (!parsed.success) {
    return { error: parsed.error.errors[0]?.message ?? '입력값이 유효하지 않습니다' }
  }

  // 2. 환경변수 확인
  const operatorEmail = process.env.OPERATOR_EMAIL
  if (!operatorEmail) {
    return { error: '수신 이메일이 설정되지 않았습니다. 관리자에게 문의해 주세요.' }
  }
  const resendKey = process.env.RESEND_API_KEY
  if (!resendKey) {
    return { error: '이메일 발송 설정이 완료되지 않았습니다. 관리자에게 문의해 주세요.' }
  }

  // 3. Resend 이메일 발송
  const { company, contact, period, message } = parsed.data
  const from = process.env.RESEND_FROM_EMAIL ?? 'danjiondo <onboarding@resend.dev>'
  const subject = `[단지온도] 광고 문의 — ${company}`
  const html = `
<h2>단지온도 광고 문의</h2>
<table style="border-collapse:collapse;width:100%">
  <tr><th style="text-align:left;padding:8px;border:1px solid #e5e7eb">업체명</th><td style="padding:8px;border:1px solid #e5e7eb">${company}</td></tr>
  <tr><th style="text-align:left;padding:8px;border:1px solid #e5e7eb">연락처</th><td style="padding:8px;border:1px solid #e5e7eb">${contact}</td></tr>
  <tr><th style="text-align:left;padding:8px;border:1px solid #e5e7eb">광고 기간</th><td style="padding:8px;border:1px solid #e5e7eb">${PERIOD_LABEL[period] ?? period}</td></tr>
  <tr><th style="text-align:left;padding:8px;border:1px solid #e5e7eb">문의 내용</th><td style="padding:8px;border:1px solid #e5e7eb">${message.replace(/\n/g, '<br>')}</td></tr>
</table>
<p style="color:#6b7280;font-size:12px">단지온도 광고 문의 시스템</p>
`

  const resend = new Resend(resendKey)
  try {
    const { error: sendErr } = await resend.emails.send({ from, to: operatorEmail, subject, html })
    if (sendErr) {
      return { error: `이메일 발송 실패: ${sendErr.message}` }
    }
  } catch (err: unknown) {
    const msg = err instanceof Error ? err.message : '알 수 없는 오류'
    return { error: `이메일 발송 중 오류가 발생했습니다: ${msg}` }
  }

  return { error: null }
}
```

import { submitAdInquiry } from '@/lib/auth/ad-inquiry-action'

        setSuccess(true)

    return (

            <input
              id="inquiry-company"
              name="company"
              className="input"
              required
              style={{ width: '100%', height: 40, fontSize: 14 }}
              placeholder="중개사무소명 또는 회사명"
            />
          </div>

          <div>
            <label
              htmlFor="inquiry-contact"
              style={{ display: 'block', font: '600 13px/1 var(--font-sans)', marginBottom: 6 }}
            >
              연락처 (이메일 또는 전화번호) <span style={{ color: 'var(--fg-negative)' }}>*</span>
            </label>
            <input

              광고 기간 <span style={{ color: 'var(--fg-negative)' }}>*</span>
            </label>
            <select
              id="inquiry-period"
              name="period"
              className="input"
              required
              defaultValue="30"
              style={{ width: '100%', height: 40, fontSize: 14 }}
            >
              <option value="7">1주 (7일)</option>
              <option value="30">1달 (30일)</option>
              <option value="90">3달 (90일)</option>

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 공개 폼 → submitAdInquiry Server Action | 비인증 사용자의 입력값. zod로 검증 필요 |
| submitAdInquiry → Resend API | 이메일 내용에 사용자 입력 포함 — HTML injection 방지 필요 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-16-02-01 | Tampering | submitAdInquiry 폼 입력값 | mitigate | zod schema: 각 필드 max 길이 제한. company≤100, contact≤200, message≤1000. 첫 번째 오류 메시지 반환 |
| T-16-02-02 | Information Disclosure | Resend 이메일 HTML 내 사용자 입력 | mitigate | 이메일 HTML에서 `\n` → `<br>` 변환 외 HTML 태그 허용 안 함. React Server Component이므로 XSS 아님. 이메일은 운영자만 수신 |
| T-16-02-03 | Denial of Service | submitAdInquiry 반복 호출 (이메일 발송 남용) | accept | MVP 규모 (월 수십 건 예상). Resend 월 3,000건 무료. 추후 rate limit 추가 가능 |
| T-16-02-04 | Information Disclosure | OPERATOR_EMAIL 환경변수 노출 | mitigate | Server Action에서만 접근 (`'use server'`). 클라이언트 번들에 포함 안 됨 |
| T-16-02-05 | Spoofing | 가짜 업체명/연락처로 문의 | accept | 운영자가 수동 검토하는 흐름. 자동화 없음. 스팸 피해 범위 제한적 |
