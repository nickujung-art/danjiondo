**B. src/lib/data/ads.ts** — update `getActiveAds`:

Replace current signature and body:
```typescript
// CRITICAL: 반드시 now() BETWEEN starts_at AND ends_at AND status='approved' 포함 (CLAUDE.md)
export async function getActiveAds(
  placement: 'banner_top' | 'sidebar' | 'in_feed' | 'map_popup',
  supabase: SupabaseClient<Database>,
  sggCode?: string,
): Promise<AdCampaign[]> {
  const now = new Date().toISOString()
  let query = supabase
    .from('ad_campaigns')
    .select('*')
    .eq('placement', placement)
    .eq('status', 'approved')
    .lte('starts_at', now)
    .gte('ends_at', now)
    .order('created_at')

  if (sggCode) {
    // Show ads targeting this sgg_code OR ads with no region restriction (null)
    query = query.or(`target_sgg_code.is.null,target_sgg_code.eq.${sggCode}`)
  }

  if (placement === 'map_popup') {
    // Only show popups that have a valid position
    query = query.not('target_lat', 'is', null).not('target_lng', 'is', null)
  }

  const { data } = await query
  return data ?? []
}
```
