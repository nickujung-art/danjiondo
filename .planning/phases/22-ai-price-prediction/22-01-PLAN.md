```typescript
// src/lib/prediction/engine.test.ts
import { describe, it, expect } from 'vitest'
import { forecast, holtwinters, doubleExp, linearForecast } from './engine'

// 테스트용 단조 상승 데이터 생성 (30개월)
function makeSeries(n: number, base = 30000, slope = 100): number[] {
  return Array.from({ length: n }, (_, i) => base + slope * i)
}

describe('forecast() 래퍼', () => {
  it('6개 미만 데이터 시 null 반환', () => {
    expect(forecast(makeSeries(5).map((p, i) => ({
      yearMonth: `2024-${String(i + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    })))).toBeNull()
  })

  it('6-11개 데이터 시 linear 모델 선택', () => {
    const data = makeSeries(8).map((p, i) => ({
      yearMonth: `2024-${String(i + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)
    expect(result?.modelName).toBe('linear')
    expect(result?.forecasts).toHaveLength(6)
  })

  it('12-23개 데이터 시 double-exp 모델 선택', () => {
    const data = makeSeries(15).map((p, i) => ({
      yearMonth: `2024-${String(i + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)
    expect(result?.modelName).toBe('double-exp')
  })

  it('24개 이상 데이터 시 holt-winters 모델 선택', () => {
    const data = makeSeries(28).map((p, i) => ({
      yearMonth: `2023-${String(i % 12 + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)
    expect(result?.modelName).toBe('holt-winters')
  })

  it('예측 6개월 반환', () => {
    const data = makeSeries(28).map((p, i) => ({
      yearMonth: `2023-${String(i % 12 + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)
    expect(result?.forecasts).toHaveLength(6)
  })

  it('신뢰구간: lower <= mean <= upper', () => {
    const data = makeSeries(28).map((p, i) => ({
      yearMonth: `2023-${String(i % 12 + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)!
    result.forecasts.forEach(f => {
      expect(f.lower).toBeLessThanOrEqual(f.mean)
      expect(f.upper).toBeGreaterThanOrEqual(f.mean)
    })
  })

  it('trainingMape가 0~1 범위', () => {
    const data = makeSeries(28).map((p, i) => ({
      yearMonth: `2023-${String(i % 12 + 1).padStart(2, '0')}`,
      avgPrice: p, txCount: 5
    }))
    const result = forecast(data)!
    expect(result.trainingMape).toBeGreaterThanOrEqual(0)
    expect(result.trainingMape).toBeLessThanOrEqual(1)
  })
})

describe('linearForecast()', () => {
  it('상승 데이터 → 양수 기울기 예측', () => {
    const prices = makeSeries(10)
    const result = linearForecast(prices, 3)
    expect(result.point[2]).toBeGreaterThan(result.point[0])
  })
})
```

**GREEN 단계: engine.ts 구현**

`src/lib/prediction/engine.ts`에 구현:
- `PricePoint` 인터페이스 (yearMonth, avgPrice, txCount)
- `PredictionResult` 인터페이스 (forecasts[], modelName, trainingMape, trainingCount)
- `linearForecast(prices: number[], horizon: number)`: 최소제곱법 선형회귀
  - 기울기 β₁ = (n·Σxy - Σx·Σy) / (n·Σx² - (Σx)²)
  - 절편 β₀ = (Σy - β₁·Σx) / n
  - 잔차 std로 신뢰구간 계산 (×1.96)
- `doubleExp(prices: number[], horizon: number)`: 이중지수평활
  - α ∈ [0.1, 0.3, 0.5, 0.7, 0.9], β ∈ [0.1, 0.3, 0.5, 0.7, 0.9] 그리드
  - RMSE 기준 최적 파라미터 선택
  - Holt's linear method: Lₜ = α·yₜ + (1-α)·(Lₜ₋₁ + Tₜ₋₁), Tₜ = β·(Lₜ - Lₜ₋₁) + (1-β)·Tₜ₋₁
- `holtwinters(prices: number[], horizon: number)`: 삼중지수평활
  - period = 12, α/β/γ 각 [0.1, 0.3, 0.5, 0.7, 0.9] 그리드 (125 조합)
  - 초기 계절 인자: 첫 번째 period의 각 월 평균 / 전체 평균
  - 예측 = (Lₙ + h·Tₙ) × Sₙ₋ₚ₊ₕ
- `forecast(data: PricePoint[], horizon = 6): PredictionResult | null`:
  - data.length < 6 → null
  - 월 문자열 생성: 입력 마지막 'YYYY-MM' + horizon개 다음 달
  - trainingMape 계산: 마지막 6개 hold-out one-step-ahead |실제-예측| / 실제 평균
  - trainingCount: data.length
