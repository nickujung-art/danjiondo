export interface GapRankingFilter {
  sggCode?: string           // URL searchParams 'sgg_code' 값
  riskLevel?: 'safe' | 'caution' | 'danger'  // URL searchParams 'risk_level' 값
}

export interface GapRankingRow {
  complexId: string
  complexName: string
  si: string | null
  gu: string | null
  sggCode: string | null
  gapRatio: number
  gapAmount: number
  jeonseRatio: number
  riskLevel: 'safe' | 'caution' | 'danger'
  saleCount: number
  jeonseCount: number
}

/**
 * complex_gap_stats JOIN complexes 조회.
 * 갭 비율 높은 순 정렬, 최대 200건.
 * createReadonlyClient()로 생성한 supabase를 전달한다.
 */
export async function getGapRankings(
  filter: GapRankingFilter,
  supabase: SupabaseClient<Database>,
): Promise<GapRankingRow[]> {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  let query = (supabase as any)
    .from('complex_gap_stats')
    .select('gap_ratio, gap_amount, jeonse_ratio, risk_level, sale_count, jeonse_count, complexes!inner(id, canonical_name, sgg_code, si, gu)')
    .order('gap_ratio', { ascending: false })
    .limit(200)

  // sgg_code 필터: 허용 목록에 있는 값만 통과
  const ALLOWED_SGG_CODES = ['48121', '48123', '48125', '48127', '48129', '48250']
  if (filter.sggCode && ALLOWED_SGG_CODES.includes(filter.sggCode)) {
    query = query.eq('complexes.sgg_code', filter.sggCode)
  }

  // risk_level 필터: 허용 목록에 있는 값만 통과
  const ALLOWED_RISK_LEVELS = ['safe', 'caution', 'danger']
  if (filter.riskLevel && ALLOWED_RISK_LEVELS.includes(filter.riskLevel)) {
    query = query.eq('risk_level', filter.riskLevel)
  }

  const { data, error } = await query

  if (error || !data) return []

  const rows = data as Array<{
    gap_ratio: string | number
    gap_amount: number
    jeonse_ratio: string | number
    risk_level: string
    sale_count: number
    jeonse_count: number
    complexes: { id: string; canonical_name: string; sgg_code: string | null; si: string | null; gu: string | null } | null
  }>

  const result: GapRankingRow[] = []
  for (const row of rows) {
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    const c = Array.isArray((row as any).complexes) ? (row as any).complexes[0] : row.complexes
    if (!c) continue
    result.push({
      complexId:   c.id,
      complexName: c.canonical_name,
      si:          c.si,
      gu:          c.gu,
      sggCode:     c.sgg_code,
      gapRatio:    Number(row.gap_ratio),
      gapAmount:   row.gap_amount,
      jeonseRatio: Number(row.jeonse_ratio),
      riskLevel:   row.risk_level as 'safe' | 'caution' | 'danger',
      saleCount:   row.sale_count,
      jeonseCount: row.jeonse_count,
    })
  }
  return result
}
