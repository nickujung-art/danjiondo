describe('computeDongChips', () => {
  it('빈 배열이면 빈 배열 반환', () => {
    expect(computeDongChips([])).toEqual([])
  })

  it('dong=null 단지는 "기타" 동으로 그룹화', () => {
    const chips = computeDongChips([makeItem({ dong: null })])
    expect(chips).toHaveLength(1)
    expect(chips[0]!.dong).toBe('기타')
  })

  it('같은 dong 단지 2개 → count=2, lat/lng 평균, maxPrice 최댓값', () => {
    const items = [
      makeItem({ lat: 35.1, lng: 128.5, dong: '상남동', recent_price: 50000 }),
      makeItem({ lat: 35.3, lng: 128.7, dong: '상남동', recent_price: 80000 }),
    ]
    const chips = computeDongChips(items)
    expect(chips).toHaveLength(1)
    expect(chips[0]!.count).toBe(2)
    expect(chips[0]!.lat).toBeCloseTo(35.2, 5)
    expect(chips[0]!.lng).toBeCloseTo(128.6, 5)
    expect(chips[0]!.maxPrice).toBe(80000)
  })

  it('gu=null 단지는 DongChip.gu도 null', () => {
    const chips = computeDongChips([makeItem({ gu: null, si: '김해시', dong: '삼계동' })])
    expect(chips[0]!.gu).toBeNull()
    expect(chips[0]!.dong).toBe('삼계동')
  })

  it('status=pre_sale 필터는 complexes.filter로 동작', () => {
    const items = [
      makeItem({ status: 'pre_sale' }),
      makeItem({ status: 'active' }),
    ]
    const preSale = items.filter(c => c.status === 'pre_sale')
    expect(preSale).toHaveLength(1)
  })

  it('서로 다른 dong 2개 → DongChip 2개 반환', () => {
    const items = [
      makeItem({ dong: '상남동' }),
      makeItem({ dong: '중앙동' }),
    ]
    const chips = computeDongChips(items)
    expect(chips).toHaveLength(2)
    const dongs = chips.map(c => c.dong).sort()
    expect(dongs).toEqual(['상남동', '중앙동'])
  })
})
