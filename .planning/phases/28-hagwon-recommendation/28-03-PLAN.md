<!-- fee_tier SQL (RESEARCH.md Code Examples — PERCENT_RANK 표준 SQL) -->
```sql
UPDATE hagwon_db h
SET fee_tier = sub.tier
FROM (
  SELECT id,
    CASE
      WHEN PERCENT_RANK() OVER (ORDER BY fee_amount DESC) <= 0.30 THEN 'premium'
      WHEN PERCENT_RANK() OVER (ORDER BY fee_amount DESC) <= 0.80 THEN 'standard'
      ELSE 'budget'
    END AS tier
  FROM hagwon_db
  WHERE fee_amount IS NOT NULL
) sub
WHERE h.id = sub.id;
```

```typescript
const defaultResult = { age_groups: [], subject_category: null, teaching_style: null }
try {
  const res = await groq.chat.completions.create({
    model: 'llama-3.1-8b-instant',
    messages: [{ role: 'user', content: buildClassifyPrompt(name, realm, crse) }],
    max_tokens: 120,
    temperature: 0.2,
    response_format: { type: 'json_object' },
  })
  const parsed = JSON.parse(res.choices[0]?.message?.content ?? '{}')
  // 유효값 검증 (Pitfall 5: 모델이 이상한 값 출력 방어)
  const VALID_AGE = ['유아','유치','초등저','초등고','중등','고등']
  const VALID_CAT = ['academic','arts','sports','language']
  const VALID_STY = ['exam_prep','enrichment','tutoring']
  return {
    age_groups: (parsed.age_groups ?? []).filter((a: string) => VALID_AGE.includes(a)),
    subject_category: VALID_CAT.includes(parsed.subject_category) ? parsed.subject_category : null,
    teaching_style: VALID_STY.includes(parsed.teaching_style) ? parsed.teaching_style : null,
  }
} catch { return defaultResult }
```

```typescript
for (let i = 0; i < withRank.length; i += 50) {

    { onConflict: 'id' }
  )
}
```
